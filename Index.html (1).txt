<script>
// Dark mode toggle
const toggleBtn = document.getElementById('darkToggle');
const downloadBtn = document.getElementById('downloadBtn');
const form = document.getElementById('contactForm');
const formMsg = document.getElementById('formMsg');
const resetBtn = document.getElementById('resetBtn');

// remember theme in localStorage
const savedTheme = localStorage.getItem('theme');
if (savedTheme === 'dark') document.documentElement.classList.add('dark');

function updateToggleText(){
const isDark = document.documentElement.classList.contains('dark');
toggleBtn.textContent = isDark ? 'Light Mode' : 'Dark Mode';
}
updateToggleText();

toggleBtn.addEventListener('click', () => {
document.documentElement.classList.toggle('dark');
const isDark = document.documentElement.classList.contains('dark');
localStorage.setItem('theme', isDark ? 'dark' : 'light');
updateToggleText();
});

// Project view buttons (simple modal via alert)
document.querySelectorAll('.viewProject').forEach(btn => {
btn.addEventListener('click', (e) => {
const title = btn.dataset.title;
const desc = btn.dataset.desc;
alert(title + '\\n\\n' + desc + '\\n\\n(Replace this with a real modal or link.)');
});
});

// Fake download CV button
downloadBtn.addEventListener('click', () => {
const cvContent = "Gayathri D\\nStudent - 2nd BSc(IT)\\nSkills: HTML, CSS, JS";
const blob = new Blob([cvContent], { type: 'text/plain' });
const url = URL.createObjectURL(blob);
const a = document.createElement('a');
a.href = url;
a.download = 'Gayathri_CV.txt';
a.click();
URL.revokeObjectURL(url);
});

// Contact form client-side validation and simulated submission
form.addEventListener('submit', (evt) => {
evt.preventDefault();
formMsg.style.color = 'var(--muted)';
const name = document.getElementById('name').value.trim();
const email = document.getElementById('email').value.trim();
const message = document.getElementById('message').value.trim();

if (!name || !email || !message) {
formMsg.textContent = 'Please fill all fields.';
return;
}
// simple email check
if (!/^[^\\s@]+@[^\\s@]+\\.[^\\s@]+$/.test(email)) {
formMsg.textContent = 'Please enter a valid email.';
return;
}

// Simulate sending
formMsg.style.color = 'green';
formMsg.textContent = 'Message sent! (simulated)';
form.reset();
});

resetBtn.addEventListener('click', () => {
form.reset();
formMsg.textContent = '';
});
</script>
